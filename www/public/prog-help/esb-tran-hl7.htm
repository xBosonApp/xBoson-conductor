  "ADT_A04": {                          // 根节点
    "$attr": {                          // 属性
      "xmlns": "urn:hl7-org:v2xml"

    "MSH": {                            // 子节点
      "MSH.1": "|",                     // 字符串型子节点

      "MSH.3": {
        "HD.1": "NES"
      },
      ... (省略)
